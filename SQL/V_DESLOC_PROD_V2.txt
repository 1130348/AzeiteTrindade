
  CREATE OR REPLACE FORCE VIEW "MEDICO"."V_DESLOC_PROD_V2" ("T_DOENTE", "DOENTE", "N_CONS", "T_EPISODIO", "EPISODIO", "PRODUTO", "PRODUTO_DESCR", "COD_SERV_ORIG", "DESCR_SERV_ORIG", "CONTACTO_SERV_ORIG", "COD_SERV", "DESCR_SERV", "CONTACTO_SERV", "DT_DESL", "USER_CRI", "DT_CRI", "FLG_POS_FINAL", "TITULO", "ABR", "NOME") AS 
  SELECT D.T_DOENTE,
  D.DOENTE,
  D.N_CONS,
  D.T_EPISODIO,
  D.EPISODIO,
  D.PRODUTO,
  DEF.NOME_CAMPO AS PRODUTO_DESCR,
  D.COD_SERV_ORIG,
  SO.DESCR_SERV AS DESCR_SERV_ORIG,
  SO.TELEFONES AS CONTACTO_SERV_ORIG,
  D.COD_SERV,
  SD.DESCR_SERV,
  SD.TELEFONES AS CONTACTO_SERV,
  D.DT_DESL,
  D.USER_CRI,
  D.DT_CRI,
  D.FLG_POS_FINAL,
  P.TITULO,
  P.ABR,
  P.NOME
FROM DESLOC_PROD_V2 D, GH.SD_PESS_HOSP_DEF P, GH.SD_SERV SO,  GH.SD_SERV SD, TAB_CAMPODEF DEF
WHERE P.USER_SYS(+) = D.USER_CRI AND 
SO.COD_SERV(+) = D.COD_SERV_ORIG AND
SD.COD_SERV(+) = D.COD_SERV AND 
DEF.CODIGO = 'PRODUTO_DESLOC' AND
DEF.VALOR = D.PRODUTO;
